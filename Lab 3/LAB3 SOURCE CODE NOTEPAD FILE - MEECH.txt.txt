#include <iostream>

using namespace std;

int main() {
	int x, y;

	cout << "Please enter an integer value : ";
	cin >> x;

	if (x >= 15)
	{
		x++;
		y = x + x - 7;
	}
	else
	{
		x = x * 2;
		y = x * x + 7;
	}

	cout << "x = " << x << " y = " << y << endl;

	system(“pause”);
	return 0;
}
